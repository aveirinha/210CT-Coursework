ADDPOLINOMIALS(p1, deg1, p2, deg2)

	IF deg1 < pdeg2
		deg3 <- deg2

		FOR i <- deg1 TO 0
			p3[i] <- p1[i] + p2[i]
		FOR i <- deg3 TO deg1+1
			p3[i] <- p2[i] 

		RETURN p3, deg3

	ELSE
		deg3 <- deg1
		FOR i <- deg2 TO 0
			p3[i] <- p1[i] + p2[i]
		FOR i <- deg3 TO deg2+1
			p3[i] <- p1[i] 

		RETURN p3, deg3
